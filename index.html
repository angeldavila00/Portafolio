// Sistema de Base de Datos - Pizzería
Total de ingresos generados por cada combo
SELECT
c.nombre AS combo, c.precio AS precio_combo, COUNT(cp.id_Producto) AS productos_incluidos, (c.precio * COUNT(cp.id_Producto)) AS ingresos_estimados
FROM Combo c LEFT JOIN Combo_Producto cp ON c.id_Combo = cp.id_Combo WHERE c.disponible = TRUE GROUP BY c.id_Combo, c.nombre, c.precio ORDER BY ingresos_estimados DESC;
